/* Switchea PC13 usando interrupt de TIM2 cada segundo*/
#include <stm32f10x.h>

void delay_ms(uint16_t t);

void TIM2_IRQHandler()
{
	TIM2->SR = 0;			//clear bandera UIF
	GPIOC->ODR ^= (1<<13); 		//switchea PC13
}

int main()
{
	RCC->APB2ENR |= 0xFC; 		//habilita reloj GPIOs
	RCC->APB1ENR |= (1<<0); 	//habilita reloj TIM2
	GPIOC->CRH = 0x44344444; 	//PC13 salida

	TIM2->PSC = 7200-1;		//PSC=7199 preescalado
	TIM2->ARR = 10000-1;		//conteos
	TIM2->SR = 0;			//clear bandera UIF
	TIM2->CR1 = 1; 			//up counting
	TIM2->DIER = (1<<0);		//habilita interrupciíon UIE	
	NVIC_EnableIRQ(TIM2_IRQn);	//habilita TIM2 interrupt
	
	GPIOB->CRL = 0x44344444;	//PB5 salida
	
	while(1)
	{
		GPIOB->ODR ^= (1<<5); 	//switchea PB5
		delay_ms(200);		//espera 0.2seg (disque)
	}
}


void delay_ms(uint16_t t)
{
	for(int i = 0; i < t; i++)
	{
		for(volatile uint16_t a = 0; a < 6000; a++)
		{}
	}
}
