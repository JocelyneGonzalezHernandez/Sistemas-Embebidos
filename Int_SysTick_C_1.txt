/* interrupción con sysTick, switchea PC13 cada seg */
#include <stm32f10x.h>

void SysTick_Handler()					//ISR generada por interrupción
{
		GPIOC->ODR ^= (1<<13);			//switcheo de PC13	
}

int main()
{
	RCC->APB2ENR = 0xFC;				//Habilita reloj GPIOs
	GPIOC->CRH = 0x44344444;			//PC13 salida

	SysTick->LOAD = 9000000-1;			//STRELOAD = (72,000,000/8)-1
	SysTick->CTRL = 0x03;		    		//Clock = AHB clock/8, int. enable, Enable = 1
	
	while(1)
	{
	}
}
